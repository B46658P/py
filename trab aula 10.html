# Separando os dados em dois grupos: alunos com acesso  internet e alunos sem acesso  internet
df_internet=df[df['IN_INTERNET']==1]
df_sem_internet=df[df['IN_INTERNET']==0]

# Fazendo uma mdia da cor/raa dos alunos com acesso  internet
df_internet=df_internet.mean()

# Fazendo uma mdia da cor/raa dos alunos sem acesso  internet
df_sem_internet=df_sem_internet.mean()

# Criando um novo dataframe com os dados das mdias
fin=pd.DataFrame({'Com acesso  internet':df_internet,'Sem acesso  internet':df_sem_internet})

# Renomeando as linhas
fin=fin.rename(index={'QT_MAT_BAS_ND':'No declarada','QT_MAT_BAS_BRANCA':'Branca','QT_MAT_BAS_PRETA':'Preta','QT_MAT_BAS_PARDA':'Parda','QT_MAT_BAS_AMARELA':'Amarela','QT_MAT_BAS_INDIGENA':'Indgena'})

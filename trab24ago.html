# Separando os dados em dois grupos: alunos com acesso à internet e alunos sem acesso à internet
df_internet=df[df['IN_INTERNET']==1]
df_sem_internet=df[df['IN_INTERNET']==0]

# Fazendo uma média da cor/raça dos alunos com acesso à internet
df_internet=df_internet.mean()

# Fazendo uma média da cor/raça dos alunos sem acesso à internet
df_sem_internet=df_sem_internet.mean()

# Criando um novo dataframe com os dados das médias
fin=pd.DataFrame({'Com acesso à internet':df_internet,'Sem acesso à internet':df_sem_internet})

# Renomeando as linhas
fin=fin.rename(index={'QT_MAT_BAS_ND':'Não declarada','QT_MAT_BAS_BRANCA':'Branca','QT_MAT_BAS_PRETA':'Preta','QT_MAT_BAS_PARDA':'Parda','QT_MAT_BAS_AMARELA':'Amarela','QT_MAT_BAS_INDIGENA':'Indígena'})
